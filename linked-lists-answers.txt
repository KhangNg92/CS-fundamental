reverse() {
    
	let current = this.head;
    
	let nextNode = current;
    
	let previous = null


    
while (current) {
      
	nextNode = current.next;
      
	current.next = previous;
      
	previous = current;
      
	current = nextNode;
    }

    
	  this.head = previous;
  }
}
